## 2. 그리디 문제 풀이 1

#### 1) 동전 0

: https://www.acmicpc.net/problem/11047

#### 내가 푼 풀이

```js
const fs = require("fs");
const filePath = process.platform === "linux" ? "/dev/stdin" : "/input.txt";
let input = fs
  .readFileSync(__dirname + filePath)
  .toString()
  .split("\n");

let num = Number(input[0].split(" ")[0]); // 동전의 개수
let val = Number(input[0].split(" ")[1]); // 만들어야 할 금액
let arr = [];

//[1, 5, 10, 50, 100, 500 ,,,,]
for (let i = 1; i <= num; i++) {
  arr.push(Number(input[i]));
}
//내림차순 정렬
arr.sort((a, b) => b - a);

let cnt = 0; // 갯수

for (let i = 0; i < arr.length; i++) {
  if (val < arr[i]) {
    continue;
  } else {
    // 갯수 담기
    cnt += parseInt(val / arr[i]);
    // 남은 동전 금액 값
    val %= arr[i];

    if (val === 0) {
      break;
    }
  }
}

console.log(cnt);
```

#### 2) ATM

: https://www.acmicpc.net/problem/11399

#### 내가 푼 풀이

```js
const fs = require("fs");
const filePath = process.platform === "linux" ? "/dev/stdin" : "/input.txt";
let input = fs
  .readFileSync(__dirname + filePath)
  .toString()
  .split("\n");

let n = Number(input[0]);
let arr = input[1].split(" ").map(Number);

// [ 1, 2, 3, 3, 4 ]
arr.sort((a, b) => a - b);

let ans = []; //
let cnt = 0; // 각 인출하는데 걸리는 시간
for (let i = 0; i < n; i++) {
  cnt += arr[i];
  ans.push(cnt); // [ 1, 3, 6, 9, 13 ]
}

let sum = 0;
for (let x of ans) {
  sum += x;
}
console.log(sum);
```

#### 3) 잃어버린 괄호

: https://www.acmicpc.net/problem/1541

#### 내가 푼 풀이

풀지 못함

#### 다른 풀이

```js
const { group } = require("console");
const fs = require("fs");
const filePath = process.platform === "linux" ? "/dev/stdin" : "/input.txt";
let input = fs
  .readFileSync(__dirname + filePath)
  .toString()
  .split("\n");

let groups = input[0].split("-");
let ans = 0;

for (let i = 0; i < groups.length; i++) {
  // 각 그룹 내부에서 덧셈 연산 적용
  let sum = groups[i]
    .split("+")
    .map(Number)
    .reduce((a, b) => a + b);
  // 각 그룹 뺄셈
  if (i == 0) {
    ans += sum;
  } else {
    ans -= sum;
  }
}
console.log(ans);
```
