## 7. 정렬 문제 풀이2

### 1) 좌표 정렬하기

: https://www.acmicpc.net/problem/11650

#### 내가 푼 풀이

```js
let input = require("fs").readFileSync("/dev/stdin").toString().split("\n");

const num = Number(input[0]);
const data = [];
for (let i = 1; i <= num; i++) {
  let [x, y] = input[i].split(" ").map(Number);
  data.push([x, y]);
}
function compare(a, b) {
  if (a[0] !== b[0]) return a[0] - b[0]; //x좌표 기준 오름차순
  else return a[1] - b[1]; //y좌표 기준 오름차순
}

data.sort(compare);

for (let i = 0; i < num; i++) {
  console.log(...data[i]);
}
```

### 3) 단어 정렬

: https://www.acmicpc.net/problem/1181

#### 내가 푼 풀이

```js
let input = require("fs").readFileSync("/dev/stdin").toString().split("\n");

const num = Number(input[0]);
const data = [];
for (let i = 1; i <= num; i++) {
  data.push(input[i]);
}

//중복제거
const value = [...new Set(data)];

function compare(a, b) {
  if (a.length !== b.length) return a.length - b.length;
  else return a.localeCompare(b);
}

value.sort(compare);

//const ans = "";
let ans = "";
for (let i = 0; i < value.length; i++) {
  ans += value[i] + "\n";
}

console.log(ans);
```

런타임 에러난다

```js
const ans = ""; => let ans = "";
```

여기를 수정해주니 해결 된다.

#### 다른 풀이

```js
let input = require("fs").readFileSync("/dev/stdin").toString().split("\n");

let num = Number(input[0]);
let arr = [];
for (let i = 1; i <= num; i++) {
  arr.push(input[i]);
}

// 중복 제거
arr = [...new Set(arr)];

arr.sort((a, b) => {
  // 수정: 화살표 함수의 문법 수정
  if (a.length != b.length) return a.length - b.length;
  else {
    if (a < b) return -1;
    else if (a > b) return 1;
    else return 0;
  }
});

for (let x of arr) {
  console.log(x);
}
```
