# 문제풀이 3

### 1. 좌표 압축(https://www.acmicpc.net/problem/18870)

```
let fs = require('fs');
let input = fs.readFileSync('/dev/stdin').toString().split('\n');

let n = Number(input[0]);
let nums = input[1].split(" ").map(Number);

let uniqueNums = [...new Set(nums)];
uniqueNums.sort((a, b) => a - b);

let mapping = new Map();

for(let i = 0; i < uniqueNums.length; i++){
    // 오름차순대로 인덱싱 맵핑
    mapping.set(uniqueNums[i], i);
}

let answer = "";

// 원본 배열 nums
// 오름차순 배열 uniqueArr의 맵핑
for(let x of nums){
    answer += mapping.get(x) + " ";
}

console.log(answer);

```

### 2. 나이순 정렬(https://www.acmicpc.net/problem/10814)

```
let fs = require('fs');
let input = fs.readFileSync('/dev/stdin').toString().split('\n');

let n = Number(input[0]);
let arr = [];

for(let i = 0; i < n; i++){
    let name = input[i+1].split(" ")[0];
    let age = input[i+1].split(" ")[1];
    arr.push([name, age]);
}

const compare = ((a, b) => a[0] - b[0]);

arr.sort(compare);

for(let x of arr){
    console.log(x[0] + " " + x[1])
}

```

### 3. 소트인사이드(https://www.acmicpc.net/problem/1427)

```
let fs = require('fs');
let input = fs.readFileSync('/dev/stdin').toString().split('\n');

let num = input[0].split('');

const compare = (a, b) => {
    if(a > b) return -1;
    else if(a < b) return 1;
    else return 0;
}

num.sort(compare);

let answer = "";

for(let x of num){
    answer += x;
}

console.log(answer);
```
