# 문제풀이 1

### 1. 동전 0(https://www.acmicpc.net/problem/11047)

```js
let fs = require("fs");
let input = fs.readFileSync("/dev/stdin").toString().split("\n");

let [n, k] = input[0].split(" ").map(Number);

let arr = [];

for (let i = 0; i < n; i++) {
  arr.push(Number(input[i + 1]));
}

let count = 0;

for (let i = arr.length - 1; i >= 0; i--) {
  if (k >= arr[i]) {
    count += parseInt(k / arr[i]);
    k = k % arr[i];
  }
}

console.log(count);
```

### 2. ATM(https://www.acmicpc.net/problem/11399)

```js
let fs = require("fs");
let input = fs.readFileSync("/dev/stdin").toString().split("\n");

let n = Number(input[0]);
let p = input[1].split(" ").map(Number);

const compare = (a, b) => {
  if (a < b) return -1;
  else return 1;
};

p.sort(compare);

let wait = 0;
let count = 0;
let answer = 0;

for (let i = 0; i < n; i++) {
  count = wait + p[i];
  answer += count;
}

console.log(answer);
```

### 3. 잃어버린 괄호(https://www.acmicpc.net/problem/1541)

```js
let fs = require("fs");
let input = fs.readFileSync("/dev/stdin").toString().split("\n");

let n = input[0];

let count = 0;

if (n.includes("-")) {
  let whole = n.split("-");

  for (let i = 0; i < whole.length; i++) {
    if (whole[i].includes("+")) {
      let plusNum = whole[i].split("+").map(Number);
      let plus = 0;

      for (let j = 0; j < plusNum.length; j++) {
        plus += plusNum[j];
      }

      whole[i] = String(plus);
    }
  }

  count = Number(whole[0]);

  for (let k = 1; k < whole.length; k++) {
    count -= Number(whole[k]);
  }

  console.log(count);
} else {
  let nums = n.split("+").map(Number);
  for (let i = 0; i < nums.length; i++) {
    count += nums[i];
  }
  console.log(count);
}
```
